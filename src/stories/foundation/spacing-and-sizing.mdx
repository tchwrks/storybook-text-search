## Best Practices

<DosDonts 
    dos={[
        {
            title: "Use Consistent Spacing",
            description: "Apply spacing tokens consistently across components",
            code: `// Good: Using spacing scale
.card {
  padding: var(--spacing-md);
  gap: var(--spacing-sm);
  margin-bottom: var(--spacing-lg);
}`
        },
        {
            title: "Scale Appropriately",
            description: "Choose spacing based on component hierarchy",
            code: `// Good: Hierarchical spacing
.section {
  margin-bottom: var(--spacing-2xl);  /* Large section gap */
}
.section__header {
  margin-bottom: var(--spacing-lg);  /* Medium component gap */
}
.section__content {
  gap: var(--spacing-md);  /* Standard content gap */
}`
        }
    ]}
    donts={[
        {
            title: "Avoid Magic Numbers",
            description: "Don't use arbitrary pixel values",
            code: `// Bad: Arbitrary spacing
.element {
  margin: 23px;  /* Arbitrary value */
  padding: 17px;  /* Not from spacing scale */
}`
        },
        {
            title: "Skip Scale Steps",
            description: "Don't skip spacing scale steps without reason",
            code: `// Bad: Inconsistent scale jumps
.container {
  gap: var(--spacing-xs);  /* 8px */
  margin: var(--spacing-2xl);  /* Jumps to 48px */
}`
        }
    ]}
/>
