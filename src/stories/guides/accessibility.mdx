## Keyboard Navigation

<DosDonts 
    dos={[
        {
            title: "Focus Management",
            description: "Ensure all interactive elements are focusable",
            code: `<button
  onClick={handleClick}
  onKeyDown={handleKeyDown}
  tabIndex={0}
>
  Click Me
</button>`
        },
        {
            title: "Focus Trapping",
            description: "Trap focus in modals and dialogs",
            code: `function Modal() {
  useFocusTrap(modalRef);
  return (
    <div ref={modalRef} role="dialog">
      {/* Modal content */}
    </div>
  );
}`
        }
    ]}
    donts={[
        {
            title: "Mouse-Only Events",
            description: "Don't rely solely on mouse events",
            code: `<div onMouseOver={showTooltip}>  // Bad
  Hover me
</div>`
        },
        {
            title: "Keyboard Traps",
            description: "Don't trap keyboard focus unnecessarily",
            code: `// Don't prevent Tab navigation
element.addEventListener('keydown', (e) => {
  e.preventDefault();  // Bad
});`
        }
    ]}
/>
